#include <ESP8266WiFi.h>
#include <FirebaseESP8266.h>
#include <DHT.h>

// WiFi Settings
#define WIFI_SSID "G54"
#define WIFI_PASSWORD "password."

// Firebase Settings
#define FIREBASE_HOST "smart-iot-app-e8912-default-rtdb.firebaseio.com"
#define FIREBASE_AUTH "AIzaSyALKUc59AoFhIGv1sX_FSiPKPbNgZxXcBA"

// Sensor Settings
#define DHTPIN D4
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);
const int soilMoisturePin = A0;

// Firebase objects
FirebaseData fbdo;
FirebaseConfig config;
FirebaseAuth auth;

void setup() {
  Serial.begin(115200);
  dht.begin();
  
  // Connect to WiFi
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  Serial.print("Connecting to WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nConnected to WiFi!");

  // Configure Firebase
  config.host = FIREBASE_HOST;
  config.signer.tokens.legacy_token = FIREBASE_AUTH;

  // Connect to Firebase
  Firebase.begin(&config, &auth);
  Firebase.reconnectWiFi(true);
  Serial.println("Firebase connected!");
}

void loop() {
  // Read sensors
  float temp = dht.readTemperature();
  float humidity = dht.readHumidity();
  int soilMoisture = analogRead(soilMoisturePin);

  // Print to Serial
  Serial.printf("Temp: %.2fÂ°C, Humidity: %.2f%%, Soil: %d\n", temp, humidity, soilMoisture);

  // Send to Firebase
  if (Firebase.setFloat(fbdo, "/temperature", temp) &&
      Firebase.setFloat(fbdo, "/humidity", humidity) &&
      Firebase.setInt(fbdo, "/soil_moisture", soilMoisture)) {
    Serial.println("Data sent to Firebase!");
  } else {
    Serial.println("Failed to send data: " + fbdo.errorReason());
  }

  delay(10000); // Wait 10 seconds
}
